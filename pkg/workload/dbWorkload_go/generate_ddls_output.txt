  w_state CHAR(2) NOT NULL

  d_zip CHAR(9) NOT NULL

--- customer ---
Table: tpcc.public.customer
 Columns:
  c_city VARCHAR(20) NOT NULL
  c_ytd_payment DECIMAL(12,2) NOT NULL
  c_payment_cnt INT8 NOT NULL
  c_id INT8 NOT NULL PRIMARY KEY
  c_first VARCHAR(16) NOT NULL
  c_middle CHAR(2) NOT NULL
  c_state CHAR(2) NOT NULL
  c_zip CHAR(9) NOT NULL
  c_phone CHAR(16) NOT NULL
  c_credit CHAR(2) NOT NULL
  c_credit_lim DECIMAL(12,2) NOT NULL
  c_w_id INT8 NOT NULL PRIMARY KEY
  c_last VARCHAR(16) NOT NULL
  c_street_1 VARCHAR(20) NOT NULL
  c_street_2 VARCHAR(20) NOT NULL
  c_since TIMESTAMP NOT NULL
  c_discount DECIMAL(4,4) NOT NULL
  c_balance DECIMAL(12,2) NOT NULL
  c_data VARCHAR(500) NOT NULL
  c_delivery_cnt INT8 NOT NULL
  c_d_id INT8 NOT NULL PRIMARY KEY
 PKs: c_w_id, c_d_id, c_id
 FKs: (c_w_id,c_d_id)→tpcc.public.district(d_w_id,d_id)

--- item ---

  i_id INT8 NOT NULL PRIMARY KEY UNIQUE
  i_im_id INT8 NULL
  i_name VARCHAR(24) NULL
  i_price DECIMAL(5,2) NULL
  i_data VARCHAR(50) NULL

--- history ---
Table: tpcc.public.history
 Columns:
  h_d_id INT8 NOT NULL
  h_c_id INT8 NOT NULL
  h_w_id INT8 NOT NULL PRIMARY KEY
  h_date TIMESTAMP NULL
  h_amount DECIMAL(6,2) NULL
  h_data VARCHAR(24) NULL
  rowid UUID NOT NULL PRIMARY KEY DEFAULT gen_random_uuid()
  h_c_d_id INT8 NOT NULL
  h_c_w_id INT8 NOT NULL
 PKs: h_w_id, rowid
 FKs: (h_c_w_id,h_c_d_id,h_c_id)→tpcc.public.customer(c_w_id,c_d_id,c_id); (h_w_id,h_d_id)→tpcc.public.district(d_w_id,d_id)
